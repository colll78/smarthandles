Smart Handles

Smart Contract support for Ada Handles

Overview

Below is an example of a user sending five ADA to the smart handle $ada-to-min.
​
​
ADA sent to this handle is routed to a DEX as a swap request. Once the swap request is submitted by the DEXs batchers, the sender will receive the some number of MIN tokens (depends on the current balance of the pool). There is a 1 ADA fee awarded to the routing agent, so the amount of MIN token the sender would receive is equivalent to the amount they would get if they swapped 14 ADA directly on the DEX UI.

How to Integrate

  • Slice the $ from the address
  • Check if the address is a script address
    • If the address is a script address, add the sender's address the the output as an inline datum.
You're all done!

export const getAdaHandle = async (assetName) => {
const network = await getNetwork();
const assetNameHex = Buffer.from(assetName).toString('hex');
if (!assetNameHex || assetNameHex.length == 0) return null;
const policy = ADA_HANDLE[network.id];
const asset = policy + assetNameHex;
const resolvedAddress = await blockfrostRequest(`/assets/${asset}/addresses`);
if (!resolvedAddress || resolvedAddress.error) return null;
return resolvedAddress[0].address;
};

import * as C from "./libs/cardano_multiplatform_lib/cardano_multiplatform_lib.generated.js";

function addressFromHexOrBech32(address: string): C.Address {
try {
return C.Address.from_bytes(fromHex(address));
} catch (_e) {
try {
return C.Address.from_bech32(address);
} catch (_e) {
throw new Error("Could not deserialize address.");
}
}
}
​
function isScriptAddress(address : string) : boolean {
try {
const parsedAddress = C.BaseAddress.from_address(
addressFromHexOrBech32(address),
)!;
return parsedAddress.payment_cred().kind() === 0 ? false : true;
} catch (_e) { /* pass */ }
try {
const parsedAddress = C.EnterpriseAddress.from_address(
addressFromHexOrBech32(address),
)!;
return parsedAddress.payment_cred().kind() === 0 ? false : true;
} catch (_e) { /* pass */ }
// Pointer Address
try {
const parsedAddress = C.PointerAddress.from_address(
addressFromHexOrBech32(address),
)!;
return parsedAddress.payment_cred().kind() === 0 ? false : true;
} catch (_e) { /* pass */ }
throw new Error("No address type matched for: " + address);
}
